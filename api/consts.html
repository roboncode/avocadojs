# Constants

A list of constants used in Orango.

# Usage

const orango = require('orango')
const { DEFAULTS, EVENTS } orango.consts

...

orango.on(EVENTS.CONNECTED, () => {
  console.log('Connected to database!')
})

await orango.connect(DEFAULTS.DATABASE)
